static std::mutex g_thread_id_mutex;
static int g_next_thread_id = 1;

// 每个线程的短ID缓存
thread_local int g_thread_short_id = 0;

int log_cur_tid() {
    if (g_thread_short_id == 0) {
        std::lock_guard<std::mutex> lock(g_thread_id_mutex);
        g_thread_short_id = g_next_thread_id++;
    }
    return g_thread_short_id;
}
