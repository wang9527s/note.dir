状态模式的定义只有一句有且只有一句话是对的，就是这句：**“允许一个对象在其内部状态改变时改变它的行为”**

#include <QCoreApplication>
#include <iostream>
#include <QDebug>
#include <unordered_map>
#include <string>
#include <vector>

// --- 定义主状态和子状态 ---

// 主状态：连接的整体阶段
enum class ConnectionMainState {
    NotConnected,     // 未连接
    Connecting,       // 连接中
    Connected,        // 连接成功
    Disconnected      // 断开连接
};

// 子状态：连接中和断开连接时的具体状态
enum class ConnectionSubState {
    None,                 // 默认状态，无操作
    Retry,                // 重试连接
    Authenticating,       // 认证阶段
    PeerConfirmation,     // 对端确认
    SelfConfirmation,     // 本端确认
    ConnectionFailed,     // 连接失败
    DisconnectRequested,  // 请求断开连接
    DisconnectedComplete  // 已断开连接
};

// --- 定义事件（用户操作） ---

enum class ConnectionAction {
    StartConnection,     // 开始连接
    RetryConnection,     // 重试连接
    ConfirmAuthentication, // 认证确认
    ConfirmPeer,         // 对端确认
    ConfirmSelf,         // 本端确认
    ConnectionSuccess,   // 连接成功
    Disconnect,          // 主动断开连接
    ConnectionFailure,   // 连接失败
    CompleteDisconnection // 完成断开连接
};

// 辅助函数：将主状态转换为字符串，方便输出
std::string to_string(ConnectionMainState state) {
    switch (state) {
    case ConnectionMainState::NotConnected: return "NotConnected";
    case ConnectionMainState::Connecting: return "Connecting";
    case ConnectionMainState::Connected: return "Connected";
    case ConnectionMainState::Disconnected: return "Disconnected";
    }
    return "Unknown";
}

// 辅助函数：将子状态转换为字符串，方便输出
std::string to_string(ConnectionSubState state) {
    switch (state) {
    case ConnectionSubState::None: return "None";
    case ConnectionSubState::Retry: return "Retry";
    case ConnectionSubState::Authenticating: return "Authenticating";
    case ConnectionSubState::PeerConfirmation: return "PeerConfirmation";
    case ConnectionSubState::SelfConfirmation: return "SelfConfirmation";
    case ConnectionSubState::ConnectionFailed: return "ConnectionFailed";
    case ConnectionSubState::DisconnectRequested: return "DisconnectRequested";
    case ConnectionSubState::DisconnectedComplete: return "DisconnectedComplete";
    }
    return "Unknown";
}

// 辅助函数：将事件转换为字符串，方便输出
std::string to_string(ConnectionAction action) {
    switch (action) {
    case ConnectionAction::StartConnection: return "StartConnection";
    case ConnectionAction::RetryConnection: return "RetryConnection";
    case ConnectionAction::ConfirmAuthentication: return "ConfirmAuthentication";
    case ConnectionAction::ConfirmPeer: return "ConfirmPeer";
    case ConnectionAction::ConfirmSelf: return "ConfirmSelf";
    case ConnectionAction::ConnectionSuccess: return "ConnectionSuccess";
    case ConnectionAction::Disconnect: return "Disconnect";
    case ConnectionAction::ConnectionFailure: return "ConnectionFailure";
    case ConnectionAction::CompleteDisconnection: return "CompleteDisconnection";
    }
    return "Unknown";

// --- 简单状态机类 ---
// 支持主状态和子状态的状态机
template <typename MainState, typename SubState, typename Event>
class MainSubStateMachine {
public:
    using TransitionTable = std::unordered_map<Event, SubState>;
    using StateMap = std::unordered_map<MainState, TransitionTable>;

    MainSubStateMachine(MainState initialMainState, SubState initialSubState)
        : currentMainState(initialMainState), currentSubState(initialSubState) {}

    // 设置主状态的转移规则
    void setTransitions(MainState mainState, const TransitionTable& transitions) {
        states[mainState] = transitions;
    }

    // 根据事件触发子状态的转移
    bool transition(Event event) {
        if (auto it = states.find(currentMainState); it != states.end()) {
            const auto& transitions = it->second;
            if (auto transIt = transitions.find(event); transIt != transitions.end()) {
                currentSubState = transIt->second;
                return true;
            }
        }
        return false;
    }

    // 获取当前主状态
    MainState getCurrentMainState() const {
        return currentMainState;
    }

    // 获取当前子状态
    SubState getCurrentSubState() const {
        return currentSubState;
    }

private:
    MainState currentMainState;
    SubState currentSubState;
    StateMap states;
};

// --- 主程序入口 ---

int main(int argc, char *argv[])
{
    QCoreApplication a(argc, argv);

    using MainState = ConnectionMainState;
    using SubState = ConnectionSubState;
    using Action = ConnectionAction;

    // 初始主状态：NotConnected，初始子状态：None
    MainSubStateMachine<MainState, SubState, Action> fsm(MainState::NotConnected, SubState::None);

    // 设置主状态为Connecting时的子状态转移规则
    fsm.setTransitions(MainState::NotConnected, {
                                                    {Action::StartConnection, SubState::Retry}
                                                });

    // 配置Connecting主状态下的子状态转移规则
    fsm.setTransitions(MainState::Connecting, {
                                                  {Action::RetryConnection, SubState::Retry},
                                                  {Action::ConfirmAuthentication, SubState::Authenticating},
                                                  {Action::ConfirmPeer, SubState::PeerConfirmation},
                                                  {Action::ConfirmSelf, SubState::SelfConfirmation}
                                              });

    fsm.setTransitions(MainState::Connecting, {
                                                  {Action::ConnectionSuccess, SubState::None},
                                                  {Action::ConnectionFailure, SubState::ConnectionFailed}
                                              });

    // 配置Disconnected主状态下的子状态转移规则
    fsm.setTransitions(MainState::Disconnected, {
                                                    {Action::Disconnect, SubState::DisconnectRequested},
                                                    {Action::CompleteDisconnection, SubState::DisconnectedComplete}
                                                });

    // 测试一组事件序列
    std::vector<Action> testEvents = {
        Action::StartConnection,  // NotConnected -> Retry
        Action::RetryConnection,  // Connecting -> Retry
        Action::ConfirmAuthentication, // Connecting -> Authenticating
        Action::ConnectionSuccess,  // Connecting -> Connected
        Action::Disconnect, // Connected -> DisconnectRequested
        Action::CompleteDisconnection  // DisconnectRequested -> DisconnectedComplete
    };

    // 执行测试事件并打印结果
    for (const auto& event : testEvents) {
        qInfo() << "事件: " << QString::fromStdString(to_string(event));
        bool success = fsm.transition(event);
        if (success) {
            qInfo() << "✅ 转移成功 -> 当前主状态: " << QString::fromStdString(to_string(fsm.getCurrentMainState()))
                    << ", 当前子状态: " << QString::fromStdString(to_string(fsm.getCurrentSubState()));
        } else {
            qInfo() << "❌ 转移失败，状态未改变 -> 当前主状态: " << QString::fromStdString(to_string(fsm.getCurrentMainState()))
                    << ", 当前子状态: " << QString::fromStdString(to_string(fsm.getCurrentSubState()));
        }
        qInfo() << "------------------------";
    }

    return a.exec();
}

#include <QCoreApplication>
#include <iostream>
#include <QDebug>
#include <unordered_map>
#include <string>
#include <memory>
#include <vector>

// --- 定义接口和状态类 ---

// 基类：每个状态的接口
class ConnectionState {
public:
    virtual ~ConnectionState() = default;
    virtual void handle() = 0; // 处理该状态下的行为
    virtual std::string getStateName() const = 0;
};

// 子状态：连接重试状态
class RetryState : public ConnectionState {
public:
    void handle() override {
        qInfo() << "Handling Retry logic";
    }
    std::string getStateName() const override {
        return "Retry";
    }
};

// 子状态：认证状态
class AuthenticatingState : public ConnectionState {
public:
    void handle() override {
        qInfo() << "Handling Authentication process";
    }
    std::string getStateName() const override {
        return "Authenticating";
    }
};

// 子状态：对端确认状态
class PeerConfirmationState : public ConnectionState {
public:
    void handle() override {
        qInfo() << "Waiting for peer confirmation";
    }
    std::string getStateName() const override {
        return "PeerConfirmation";
    }
};

// 子状态：本端确认状态
class SelfConfirmationState : public ConnectionState {
public:
    void handle() override {
        qInfo() << "Confirming connection from our side";
    }
    std::string getStateName() const override {
        return "SelfConfirmation";
    }
};

// 子状态：连接成功状态
class ConnectedState : public ConnectionState {
public:
    void handle() override {
        qInfo() << "Connection established successfully!";
    }
    std::string getStateName() const override {
        return "Connected";
    }
};

// 子状态：连接失败状态
class ConnectionFailedState : public ConnectionState {
public:
    void handle() override {
        qInfo() << "Connection failed, retrying...";
    }
    std::string getStateName() const override {
        return "ConnectionFailed";
    }
};

// 子状态：已断开状态
class DisconnectedState : public ConnectionState {
public:
    void handle() override {
        qInfo() << "Connection disconnected.";
    }
    std::string getStateName() const override {
        return "Disconnected";
    }
};

// --- 定义主状态类 ---
class Workflow : public ConnectionState {
public:
    // 主状态初始化为连接中（即启动时为 "Retry" 状态）
    Workflow()
        : currentSubState(std::make_shared<RetryState>()) {}

    // 设置当前子状态
    void setSubState(std::shared_ptr<ConnectionState> subState) {
        currentSubState = subState;
    }

    void handle() override {
        // 调用当前子状态的行为
        currentSubState->handle();
    }

    std::string getStateName() const override {
        return "workflow: " + currentSubState->getStateName();
    }

private:
    std::shared_ptr<ConnectionState> currentSubState; // 当前子状态
};

// --- 定义事件和主程序 ---

enum class ConnectionAction {
    StartConnection,     // 开始连接
    RetryConnection,     // 重试连接
    ConfirmAuthentication, // 认证确认
    ConfirmPeer,         // 对端确认
    ConfirmSelf,         // 本端确认
    ConnectionSuccess,   // 连接成功
    Disconnect,          // 主动断开连接
    ConnectionFailure,   // 连接失败
    CompleteDisconnection // 完成断开连接
};

std::string to_string(ConnectionAction action) {
    switch (action) {
    case ConnectionAction::StartConnection: return "StartConnection";
    case ConnectionAction::RetryConnection: return "RetryConnection";
    case ConnectionAction::ConfirmAuthentication: return "ConfirmAuthentication";
    case ConnectionAction::ConfirmPeer: return "ConfirmPeer";
    case ConnectionAction::ConfirmSelf: return "ConfirmSelf";
    case ConnectionAction::ConnectionSuccess: return "ConnectionSuccess";
    case ConnectionAction::Disconnect: return "Disconnect";
    case ConnectionAction::ConnectionFailure: return "ConnectionFailure";
    case ConnectionAction::CompleteDisconnection: return "CompleteDisconnection";
    }
    return "Unknown";
}

// --- 主程序入口 ---

int main(int argc, char *argv[])
{
    QCoreApplication a(argc, argv);

    // 创建主状态
    Workflow workflow;

    // 测试事件序列
    std::vector<ConnectionAction> testEvents = {
        ConnectionAction::StartConnection,  // 开始连接 -> Retry
        ConnectionAction::RetryConnection,  // 重试连接 -> Retry
        ConnectionAction::ConfirmAuthentication, // 认证确认 -> Authenticating
        ConnectionAction::ConnectionSuccess,  // 连接成功 -> Connected
        ConnectionAction::Disconnect, // 断开连接 -> Disconnected
        ConnectionAction::CompleteDisconnection  // 完成断开 -> DisconnectedComplete
    };

    // 执行测试事件并打印结果
    for (const auto& event : testEvents) {
        qInfo() << "事件: " << QString::fromStdString(to_string(event));

        // 根据事件处理对应的子状态
        switch (event) {
        case ConnectionAction::StartConnection:
            workflow.setSubState(std::make_shared<RetryState>());
            break;
        case ConnectionAction::RetryConnection:
            workflow.setSubState(std::make_shared<RetryState>());
            break;
        case ConnectionAction::ConfirmAuthentication:
            workflow.setSubState(std::make_shared<AuthenticatingState>());
            break;
        case ConnectionAction::ConnectionSuccess:
            workflow.setSubState(std::make_shared<ConnectedState>());
            break;
        case ConnectionAction::Disconnect:
            workflow.setSubState(std::make_shared<DisconnectedState>());
            break;
        case ConnectionAction::CompleteDisconnection:
            workflow.setSubState(std::make_shared<DisconnectedState>());
            break;
        default:
            break;
        }

        // 处理当前状态的行为
        workflow.handle();
        qInfo() << "当前状态: " << QString::fromStdString(workflow.getStateName());
        qInfo() << "------------------------";
    }

    return a.exec();
}
