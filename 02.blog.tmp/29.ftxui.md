```bash
#!/bin/bash

set -e

if [ ! -d "./FTXUI" ]; then
    echo "download FTXUI"
    git clone https://github.com/ArthurSonzogni/FTXUI.git
    cd FTXUI/
    git checkout v4.1.1

    rm -rf build && mkdir build && cd build

    cmake .. -DBUILD_EXAMPLES=OFF -DBUILD_TESTS=OFF
    make -j4
fi

g++ demo.cpp -std=c++20 -pthread \
    -I ./FTXUI/include   \
    ./FTXUI/build/libftxui-component.a \
    ./FTXUI/build/libftxui-dom.a   \
    ./FTXUI/build/libftxui-screen.a   \
    -o demo

```

```cpp
#include <ftxui/component/component.hpp>
#include <ftxui/component/screen_interactive.hpp>
#include <ftxui/dom/elements.hpp>
#include <iostream>
#include <thread>
#include <chrono>
#include <mutex>
#include <string>
#include <vector>

using namespace ftxui;

// 日志缓冲
std::string logs;
std::mutex logs_mutex;

void LoggerThread() {
    for (int i = 1; i <= 10; i++) {
        {
            std::lock_guard<std::mutex> lock(logs_mutex);
            logs += "Log line " + std::to_string(i) + "\n";
        }
        std::this_thread::sleep_for(std::chrono::milliseconds(500));
    }
    {
        std::lock_guard<std::mutex> lock(logs_mutex);
        logs += "✅ Logger finished.\n";
    }
}

int main() {
    auto screen = ScreenInteractive::TerminalOutput();

    // 复选框
    bool featureA = false;
    bool featureB = true;
    auto checkboxA = Checkbox("Feature A", &featureA);
    auto checkboxB = Checkbox("Feature B", &featureB);
    auto checkboxes = Container::Horizontal({checkboxA, checkboxB});

    // 单选框
    std::vector<std::string> options = {"Option 1", "Option 2", "Option 3"};
    int selected = 0;
    auto radiobox = Radiobox(&options, &selected);

    // 输入框
    std::string input_value;
    auto input = Input(&input_value, "Type here...");

    // 按钮
    std::string status = "Waiting...";
    auto button_ok = Button("OK", [&] {
        status = "Pressed OK, input=" + input_value;
        std::lock_guard<std::mutex> lock(logs_mutex);
        logs += "[User] Pressed OK\n";
    });
    auto button_cancel = Button("Cancel", [&] {
        status = "Cancelled!";
        std::lock_guard<std::mutex> lock(logs_mutex);
        logs += "[User] Pressed Cancel\n";
    });

    auto buttons = Container::Horizontal({button_ok, button_cancel});

    // 文本输入 + 按钮放同一行
    auto input_buttons = Container::Horizontal({input, buttons});

    // 总容器
    auto controls = Container::Vertical({checkboxes, radiobox, input_buttons});

    auto renderer = Renderer(controls, [&] {
        std::string logs_copy;
        {
            std::lock_guard<std::mutex> lock(logs_mutex);
            logs_copy = logs;
        }

        return vbox({
            text("================  FTXUI Demo  ================") | bold | center,
            separator(),
            // 上方控件
            vbox({
                hbox({checkboxA->Render(), checkboxB->Render()}),
                radiobox->Render(),
                hbox({input->Render(), button_ok->Render(), button_cancel->Render()}),
                text("Status: " + status) | color(Color::Yellow)
            }) | border,
            // 下方日志
            vbox({
                text("Logs:"),
                text(logs_copy) | vscroll_indicator | frame | size(HEIGHT, LESS_THAN, 10)
            }) | border
        }) | center;
    });

    // 启动日志线程
    std::thread logger(LoggerThread);

    screen.Loop(renderer);
    logger.join();

    return 0;
}
```
